public class Solution {
    public ListNode detectCycle(ListNode head) {
        if(head==null || head.next==null)return null;

        ListNode slow=head;
        ListNode fast=head;
        ListNode e=head;

        while(fast.next!=null && fast.next.next!=null)
        {
            slow=slow.next;
            fast=fast.next.next;
            if(slow==fast)
            {
                //ListNode e=head;
                while(e!=slow)
                {
                
                    slow=slow.next;
                    e=e.next;
                }
            
            return e;
            }
        }
        
        return null;
    }
}
